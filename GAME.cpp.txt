#include <iostream>
#include <cstdlib>
#include <ctime>
#include <string>
#include <locale>

using namespace std;

int main() {
    srand(time(NULL));

    string choices[] = {"rock", "paper", "scissors"};

    while (true) {
        cout << "Enter your choice (rock, paper, or scissors): ";
        string user;
        getline(cin, user);
        for (char &c : user) {
            c = tolower(c);
        }

        if (user != "rock" && user != "paper" && user != "scissors") {
            cout << "Invalid input. Please enter rock, paper, or scissors." << endl;
            continue;
        }

        int computerArr = rand() % 3;
        string computer = choices[computerArr];
        cout << "Computer's choice: " << computer << endl;

        if (user == computer) {
            cout << "It's a tie!" << endl;
        } else if ((user == "rock" && computer == "scissors") ||
                   (user == "paper" && computer == "rock") ||
                   (user == "scissors" && computer == "paper")) {
            cout << "You win!" << endl;
        } else {
            cout << "Computer wins!" << endl;
        }

        cout << "Do you want to play again? (yes/no): ";
        string again;
        getline(cin, again);
        for (char &c : again) {
            c = tolower(c);
        }
        if (again != "yes") {
            break;
        }
    }

    return 0;
}
